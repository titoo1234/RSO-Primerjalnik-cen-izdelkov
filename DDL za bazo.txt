CREATE TABLE Izdelki (
    [Ime izdelka] TEXT    NOT NULL,
    id            INTEGER PRIMARY KEY AUTOINCREMENT
);
CREATE TABLE Trgovine (
    id  INTEGER PRIMARY KEY AUTOINCREMENT,
    ime TEXT    NOT NULL
                UNIQUE
);
CREATE TABLE Cene (
    izdelek  INTEGER REFERENCES Izdelki (id),
    trgovina INTEGER REFERENCES Trgovine (id),
    cena     REAL,
    UNIQUE (
        izdelek,
        trgovina
    )
);
